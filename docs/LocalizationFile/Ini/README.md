# Ini

[Type:ConsoleApp1.MyController]
Key:Success      = Success
Key:Error        = Error (Code=0x{0:X8})

[Culture:en:Type:ConsoleApp1.MyController]
Key:Success      = Success
Key:Error        = Error (Code=0x{0:X8})

[Culture:fi:Type:ConsoleApp1.MyController]
Key:Success      = Onnistui
